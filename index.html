Friday the 13ths function (JavaScript)

The function fridayTheThirteenths accepts a start year and an end year (inclusive), and returns all of the dates where the 13th of a month lands on a Friday in the given range of year(s). You can see this kata here.

function fridayTheThirteenths(start, end) {
  let answer = [];
  end = end || start;
  for (let year = start; year <= end; year++) {
    for (let month = 1; month <= 12; month++) {
      if (new Date(year, month - 1, 13).getDay() == 5) {
        answer.push(month + "/13/" + year);
      }
    }
  }
  return answer.join(" ");
}
